userデータベース
create table user( user_id varchar(16) primary key not null, 
 password varchar(16) not null, user_name varchar(16) not null,
 language char(16) not null,  purpose char(11) not null,
 career varchar(20) not null, certification char(14) not null);


articleデータベース
create table article (
 article_id int auto_increment primary key not null,
 article_title varchar (50) not null,
 user_id varchar (16) not null,
 article_create timestamp not null,
 article_update timestamp not null,
 article_language char (15) not null,
 article_purpose char (10) not null,
 article_career char (5) not null,
 article_certification char (13) not null,
 article_favs int not null,
 article_link varchar (20) not null
);


commentデータベース
create table comment (
 comment_id int auto_increment primary key not null,
 article_id int not null,
 user_id varchar (16) not null,
 comment_date timestamp not null,
 comment_text varchar (200) not null
);


communityデータベース
create table community (
 community_id int auto_increment primary key not null,
 community_date timestamp not null,
 community_name varchar (30) not null,
 community_language char (15) not null,
 community_purpose char (10) not null,
 community_career char (5) not null,
 community_certification char (13) not null,
 community_summary varchar (100) not null
);


memberデータベース
create table member (
 participation_id int auto_increment primary key not null
 community_id (int) not null,
 user_id varchar (16) not null
);


chatデータベース
create table chat (
 remark_id int auto_increment primary key not null,
 community_id int not null,
 user_id varchar (16) not null,
 remark_text varchar (500) not null,
 remark_date timestamp not null
);
